SQL queries:

# Use sqlcmd to test connection to your SQL Server
sqlcmd -S my-sql-server01.database.windows.net -U sqladmin -P Password1234! -Q "SELECT @@VERSION"

create table:
sqlcmd -S my-sql-server01.database.windows.net -U sqladmin -P Password1234! -d mydatabase -Q "CREATE TABLE VisitorLogs (ID INT IDENTITY(1,1) PRIMARY KEY, VisitorIP VARCHAR(15), VisitDateTime DATETIME DEFAULT GETDATE());"


insert rows:
sqlcmd -S my-sql-server01.database.windows.net -U sqladmin -P Password1234! -d mydatabase -Q "INSERT INTO VisitorLogs (VisitorIP) VALUES ('192.168.1.1'), ('192.168.1.2'), ('192.168.1.3');"


list table rows:
sqlcmd -S my-sql-server01.database.windows.net -U sqladmin -P Password1234! -d mydatabase -Q "SELECT * FROM VisitorLogs;"
